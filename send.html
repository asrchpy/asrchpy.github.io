def scroll_down() -> None:
    """Scrolls page until it reaches the end"""

    last_height = driver.execute_script("return document.body.scrollHeight")

    while True:
        driver.execute_script(
            "window.scrollTo(0, document.body.scrollHeight);"
        )

        time.sleep(2)

        new_height = driver.execute_script(
            "return document.body.scrollHeight"
        )

        if new_height == last_height:
            break

        last_height = new_height

def search_ddg(url: str) -> list[str]:
    """Returns results from DuckDuckGo using a driver"""

    output: list[str] = []
    with webdriver.Firefox(options=options) as driver:

        try:
            driver.get(url2)

            logger.debug("Loooking for search bar")
            text_area = WebDriverWait(driver, 10).until(
                EC.visibility_of_element_located((By.CLASS_NAME, "query"))
            )

            time.sleep(random.randrange(0, 2))
            text_area.send_keys(query)

            time.sleep(random.randrange(0, 2))
            text_area.send_keys(Keys.ENTER)

            scroll_down()

            lnks = driver.find_elements(By.XPATH, "//a[@href]")

            for lnk in lnks:
                urlStr = lnk.get_attribute("href")
                normalizedOutput = url_normalize(urlStr)

                blackList: list[str] = [
                    "spreadprivacy",
                    "javascript:",
                    "bing",
                    "bingj",
                    "yahoo",
                    "google",
                    "duckduckgo",
                ]

                if not any(val in urlStr for val in blackList):
                    output.append(f"[DDG] {urlStr}\n")

        except NoSuchElementException as e:
            logger.error(e)
            logger.error("No element\n\t'-> continuing")

        except ElementNotVisibleException as e:
            logger.error(e)
            logger.error("Element not visible\n\t'-> continuing")

        except TimeoutException as e:
            logger.error(e)
            logger.error("Timed out\n\t'-> continuing")
    return output

def search_google(url: str) -> list[str]:
    """Returns search results from google using a driver"""

    output: list[str] = []
    with webdriver.Firefox(options=options) as driver:
        """
        Opens Google and searches query given by user
        """

        def scroll_down():
            last_height = driver.execute_script(
                "return document.body.scrollHeight"
            )

            while True:
                driver.execute_script(
                    "window.scrollTo(0, document.body.scrollHeight);"
                )

                time.sleep(2)

                new_height = driver.execute_script(
                    "return document.body.scrollHeight"
                )

                if new_height == last_height:
                    break

                last_height = new_height

        logger.debug(f"Sending {query}")
        logger.debug("Opening Driver")

        driver.get(url1)
        time.sleep(2)

        """Find search box element text_area
        Send query to text_area element
        """

        try:
            text_area = WebDriverWait(driver, 10).until(
                EC.visibility_of_element_located((By.CLASS_NAME, "gLFyf"))
            )

            time.sleep(random.randrange(0, 2))

            text_area.send_keys(query)

            time.sleep(random.randrange(0, 2))
            text_area.send_keys(Keys.ENTER)

            scroll_down()

            displ = driver.find_elements(By.TAG_NAME, "cite")
            lnks = driver.find_elements(By.XPATH, "//a[@href]")

            for lnk, dis in zip(lnks, displ):
                urlStr = lnk.get_attribute("href")
                disStr = dis.text

                normalizedOutput = url_normalize(urlStr)
                blackList: list[str] = [
                    "spreadprivacy",
                    "javascript:",
                    "bing",
                    "bingj",
                    "yahoo",
                    "google",
                    "duckduckgo",
                ]

                if not any(val in disStr for val in blackList) and not any(
                    val in urlStr for val in blackList
                ):
                    output.append(f"[GGL] {urlStr}\n")

            logger.debug("Driver quit")
            logger.debug("Deleted all cookies")

        except NoSuchElementException as e:
            logger.error(e)
            logger.error("No element\n\t'-> continuing")

        except ElementNotVisibleException as e:
            logger.error(e)
            logger.error("Element not visible\n\t'-> continuing")

        except TimeoutException as e:
            logger.error(e)
            logger.error("Timed out\n\t'-> continuing")
    return output


def search_google_nodriver(url: str) -> list[str]:
    """Returns search results from Google without opening a driver"""
    output: list[str] = []
    url_google_nd: str = "http://www.google.com/search?q=" + query
    r = requests.get(url_google_nd)
    html_content: str = r.text
    soup = BeautifulSoup(html_content, "lxml")

    links: list = [a.get("href") for a in soup.find_all("a", href=True)]

    for elem in links:
        if "//support.google.com/websearch/answer/86640" in elem:
            logging.error(
                "Captcha detected. Please change your IP or wait a few minutes"
            )
        if (
            "google" not in elem
            and "/advanced" not in elem
            and "/search?" not in elem
            and "/?sa=" not in elem
        ):
            head, sep, tail = elem.partition("&sa=")
            output.append("[GGLnd] " + head.replace("/url?q=", ""))
            
    return(output)

def search_ddg_nodriver(url: str) -> list[str]:
    """Returns search results from DuckDuckGo without opening a driver"""
    output: list[str] = []
    url_ddg_nd: str = "https://duckduckgo.com/html/?q=" + query
    headers: dict[str, str] = {
        "User-Agent": "Mozilla/5.0 (Macintosh; Intel Mac OS X 10.15; rv:84.0) Gecko/20100101 Firefox/84.0",
    }

    r = requests.get(url_ddg_nd)
    page = requests.get("https://duckduckgo.com/html/?q=test", headers=headers).text
    soup = BeautifulSoup(page, "html.parser").find_all(
        "a", class_="result__url", href=True
    )
    for link in soup:
        output.append("[DDGnd] ", link["href"])

    return output
